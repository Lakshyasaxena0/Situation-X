export type IntentType =
  | "decision"
  | "relationship"
  | "conflict"
  | "career"
  | "health"
  | "unclear";

export type IntentResult = {
  intent: IntentType;
  confidence: "low" | "medium" | "high";
  score: number;
};

const KEYWORDS: Record<IntentType, string[]> = {
  decision: [
    "should i",
    "kya karu",
    "what should",
    "decide",
    "choose",
    "option",
    "select",
    "konsa",
  ],
  relationship: [
    "love",
    "relationship",
    "girlfriend",
    "boyfriend",
    "partner",
    "marriage",
    "breakup",
    "ex",
    "ladki",
    "ladka",
  ],
  conflict: [
    "fight",
    "argument",
    "problem",
    "issue",
    "misunderstanding",
    "clash",
    "dispute",
    "ladhai",
  ],
  career: [
    "job",
    "career",
    "work",
    "salary",
    "promotion",
    "college",
    "study",
    "future",
  ],
  health: [
    "health",
    "pain",
    "stress",
    "anxiety",
    "disease",
    "illness",
  ],
  unclear: [],
};

// Normalize text
function normalize(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^\w\s]/gi, "")
    .replace(/\s+/g, " ")
    .trim();
}

// Count keyword matches
function countMatches(text: string, keywords: string[]): number {
  let count = 0;
  for (const keyword of keywords) {
    if (text.includes(keyword)) {
      count++;
    }
  }
  return count;
}

// Score calculation
function calculateScores(text: string): Record<IntentType, number> {
  const scores: Record<IntentType, number> = {
    decision: 0,
    relationship: 0,
    conflict: 0,
    career: 0,
    health: 0,
    unclear: 0,
  };

  for (const intent in KEYWORDS) {
    const key = intent as IntentType;
    scores[key] = countMatches(text, KEYWORDS[key]);
  }

  return scores;
}

// Confidence calculation
function getConfidence(score: number): "low" | "medium" | "high" {
  if (score >= 4) return "high";
  if (score >= 2) return "medium";
  return "low";
}

// MAIN FUNCTION
export function analyzeIntent(input: string): IntentResult {
  if (!input || input.length < 10) {
    return {
      intent: "unclear",
      confidence: "low",
      score: 0,
    };
  }

  const normalized = normalize(input);
  const scores = calculateScores(normalized);

  // Find max score
  let maxIntent: IntentType = "unclear";
  let maxScore = 0;

  for (const intent in scores) {
    const key = intent as IntentType;
    if (scores[key] > maxScore) {
      maxScore = scores[key];
      maxIntent = key;
    }
  }

  // Edge case: no matches
  if (maxScore === 0) {
    return {
      intent: "unclear",
      confidence: "low",
      score: 0,
    };
  }

  return {
    intent: maxIntent,
    confidence: getConfidence(maxScore),
    score: maxScore,
  };
}
